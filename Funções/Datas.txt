SELECT CURDATE(); /* mostra a data atual*/

SELECT CURRENT_TIME(); /* mostra a hora atual*/

SELECT CURRENT_TIMESTAMP(); /* mostra a data e a hora atual*/

SELECT YEAR(CURRENT_TIMESTAMP()); /* mostra o ano atual*/

SELECT DAY(CURRENT_TIMESTAMP()); /* mostra o dia atual*/

SELECT MONTH(CURRENT_TIMESTAMP()); /* mostra o mês atual*/

SELECT MONTHNAME(CURRENT_TIMESTAMP()); /* mostra o nome do mês atual*/

SELECT DATEDIFF(CURRENT_TIMESTAMP(), '2022-01-01') AS RESULTADO; /*contabiliza o intervalo entre datas*/

SELECT DATEDIFF(CURRENT_TIMESTAMP(), '1993-02-07') AS RESULTADO;

SELECT CURRENT_TIMESTAMP() AS DIA_HOJE, DATE_SUB(CURRENT_TIMESTAMP(), INTERVAL 5 DAY) AS RESULTADO;

SELECT DISTINCT DATA_VENDA, DAYNAME(DATA_VENDA) AS DIA, MONTHNAME(DATA_VENDA) AS MES, YEAR(DATA_VENDA) AS ANO FROM notas_fiscais;

SELECT NOME, TIMESTAMPDIFF (YEAR, DATA_DE_NASCIMENTO, CURDATE()) AS IDADE FROM  tabela_de_clientes;
