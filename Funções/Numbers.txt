SELECT ROUND(12.53333332323) AS RESULTADO; /* Arredonda valores para cima*/

SELECT FLOOR(12.33333332323) AS RESULTADO; /* Arredonda valores para baixo*/

SELECT RAND() AS RESULTADO; /* Traz um número aleatório*/

SELECT NUMERO, QUANTIDADE, PRECO, ROUND(QUANTIDADE * PRECO, 2) AS FATURAMENTO FROM itens_notas_fiscais;

SELECT YEAR(DATA_VENDA), FLOOR(SUM(IMPOSTO * (QUANTIDADE * PRECO))) AS IMPOSTO
FROM notas_fiscais NF
INNER JOIN itens_notas_fiscais INF ON NF.NUMERO = INF.NUMERO
WHERE YEAR(DATA_VENDA) = 2016
GROUP BY YEAR(DATA_VENDA)
